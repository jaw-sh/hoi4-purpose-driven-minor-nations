# Resources (use in conjunction with any/every limits)
can_build_aluminium = {
    OR = {
        is_african = yes
        is_asian = yes
        is_australian = yes
        is_south_american = yes
    }
}
can_build_steel = {
    always = yes
}
can_build_oil = {
    OR = {
        is_coastal = yes
        is_middle_eastern = yes
    }
}
can_build_rubber = {
    OR = {
        is_african = yes
        is_asian = yes
        is_south_american = yes
    }
}
can_build_tungsten = {
    always = yes
}
can_build_chromium = {
    always = yes
}
